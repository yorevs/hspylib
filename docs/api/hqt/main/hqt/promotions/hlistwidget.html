 48    def item_changed(self, item: QListWidgetItem):
 49        """Handles itemChanged event to avoid adding duplicates to the list."""
 50        existing = self.findItems(item.text(), Qt.MatchFixedString)
 51        if len(existing) > 1:
 52            self.del_item(item)

 54    def addItem(self, item: QListWidgetItem) -> None:
 55        """Adds the specified element to this list regardless if it is present or not."""
 56        prev = self.item(self.count() - 1)
 57        if prev:
 58            item.setFlags(prev.flags())
 59        super().addItem(item)
 60        self._items.append(item)
 61
 62    def is_empty(self) -> bool:
 63        """Returns true if this list contains no elements."""
 64        return len(self._items) == 0
 65
 66    def set_item(self, item: Union[str, QListWidgetItem], flags: Union[Qt.ItemFlags, Qt.ItemFlag] = None) -> None:
 67        """Adds the specified element to this list if it is not already present."""
 68        if not self.findItems(item, Qt.MatchFixedString):
 69            if isinstance(item, str):
 70                w_item = QListWidgetItem(item)
 71                w_item.setFlags(flags or Qt.ItemIsSelectable | Qt.ItemIsEditable | Qt.ItemIsEnabled)
 72                self.addItem(w_item)
 73            else:
 74                item.setFlags(flags or Qt.ItemIsSelectable | Qt.ItemIsEditable | Qt.ItemIsEnabled)
 75                self.addItem(item)
 76
 77    def remove_item_at_index(self, row: int) -> Optional[QListWidgetItem]:
 78        """Removes the element at the specified row from this list if it is present."""
 79        item = self.takeItem(row)
 80        if item:
 81            self._items.remove(self._items[row])
 82        return item
 83
 84    def del_item(self, item_or_index: Union[int, QModelIndex, QListWidgetItem]) -> Optional[QListWidgetItem]:
 85        """Removes the specified element from this list if it is present."""
 86        item = None
 87        if isinstance(item_or_index, int):
 88            if 0 <= item_or_index < len(self._items):
 89                item = self.remove_item_at_index(item_or_index)
 90        elif isinstance(item_or_index, QListWidgetItem):
 91            index = self.indexFromItem(item_or_index)
 92            if index:
 93                item = self.remove_item_at_index(index.row())
 94        elif isinstance(item_or_index, QModelIndex):
 95            item = self.remove_item_at_index(item_or_index.row())
 96
 97        return item

125    def set_editable(self, editable: bool = True) -> None:
126        """Set editable property."""
127        for item in self._items:
128            flags = int(item.flags())
129            if editable:
130                item.setFlags(flags | Qt.ItemIsEditable)
131            else:
132                item.setFlags(self.unset_flag(flags, int(Qt.ItemIsEditable)))
133        self._editable = editable
134
135    def set_selectable(self, selectable: bool = True) -> None:
136        """Set selectable property."""
137        for item in self._items:
138            flags = int(item.flags())
139            if selectable:
140                item.setFlags(flags | Qt.ItemIsSelectable)
141            else:
142                item.setFlags(self.unset_flag(flags, int(Qt.ItemIsSelectable)))
143        self._selectable = selectable
144
145    def _context_menu(self) -> None:
146        """Display the custom context menu."""
147        if self._context_menu_enable:
148            ctx_menu = QMenu(self)
149            if self._editable:
150                ctx_menu.addAction("Add Item", lambda: self.set_item("<new_item>"))
151                if not self.is_empty():
152                    ctx_menu.addAction("Delete Item", lambda: self.del_item(self.currentIndex().row()))
153                    ctx_menu.addSeparator()
154                    ctx_menu.addAction("Clear list", self.clear)
155
156            for act in self._custom_menu_actions:
157                check_not_none(act)
158                check_state(len(act) == 3, f"Invalid custom menu action: {act}")
159                check_argument(callable(act[1]), "The action must be callable")
160                if act[2]:
161                    ctx_menu.addSeparator()
162                ctx_menu.addAction(act[0], act[1])
163
164            ctx_menu.exec_(QCursor.pos())
