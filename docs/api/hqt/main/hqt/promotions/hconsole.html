 66    def line_number_area_paint_event(self, event: QPaintEvent) -> None:
 67        """Paint the numbers of the line number area"""
 68        margin_right = -2
 69        bg_color = self.palette().color(QPalette.Window).lighter(100)
 70        txt_color = self.palette().color(QPalette.WindowText).darker(160)
 71        painter = QPainter(self._line_number_area)
 72        painter.fillRect(event.rect(), bg_color)
 73        block = self.firstVisibleBlock()
 74        block_number = block.blockNumber()
 75        top = round(self.blockBoundingGeometry(block).translated(self.contentOffset()).top())
 76        bottom = top + round(self.blockBoundingRect(block).height())
 77        while block.isValid() and top <= event.rect().bottom():
 78            if block.isVisible() and bottom >= event.rect().top():
 79                number = str(block_number + 1)
 80                painter.setPen(txt_color)
 81                painter.drawText(
 82                    margin_right,
 83                    top,
 84                    self._line_number_area.width(),
 85                    self.fontMetrics().height(),
 86                    Qt.AlignRight,
 87                    number,
 88                )
 89            block = block.next()
 90            top = bottom
 91            bottom = top + round(self.blockBoundingRect(block).height())
 92            block_number += 1
 93
 94    def line_number_area_width(self) -> int:
 95        """Return the current line number area width"""
 96        if self._show_line_numbers:
 97            digits = 1
 98            margin_left = 10
 99            max_count = max(1, self.document().blockCount())
100            char_width = self.fontMetrics().widthChar("_")
101            # Check the amount of digits to be printed
102            while max_count >= 10:
103                max_count /= 10
104                digits += 1
105
106            return margin_left + char_width * digits
107
108        return 0
