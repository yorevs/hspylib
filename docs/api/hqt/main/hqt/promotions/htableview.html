 29class HTableView(QTableView):

 32    def __init__(self, parent: Optional[QWidget], placeholder: Optional[str] = None):
 33        super().__init__(parent)
 34        self._context_menu_enable = True
 35        self._copyable = True
 36        self._clearable = True
 37        self._deletable = True
 38        self._placeholder = placeholder or "No data to display"
 39        self._custom_menu_actions = []

 42        self.horizontalHeader().setSectionResizeMode(QHeaderView.ResizeToContents)

 47    def paintEvent(self, event: QPaintEvent) -> None:
 48        super().paintEvent(event)
 49        if self.model() is not None and self.model().rowCount() > 0:
 50            return
 51        color = self.palette().placeholderText().color()
 52        painter = QPainter(self.viewport())
 53        painter.save()
 54        painter.setPen(color)
 55        elided_text = self.fontMetrics().elidedText(self._placeholder, Qt.ElideRight, self.viewport().width())
 56        painter.drawText(self.viewport().rect(), Qt.AlignCenter, elided_text)
 57        painter.restore()
 58
 59    def filters(self) -> Optional[CollectionFilter]:
 60        """Return current data filters"""
 61        return self.model().filters() if self.model() else None
 62
 63    def refresh(self) -> None:
 64        """Synchronize view and model data"""
 65        self.model().refresh_data()
 66
 67    def clear(self):
 68        """Clear the entire table"""
 69        model = self.model()
 70        if model:
 71            model.clear()
 72
 73    def copy(self) -> None:
 74        """Copy selected cell into clipboard"""
 75        sel_model = self.selectionModel()
 76        if sel_model:
 77            index_list = sel_model.selectedIndexes()
 78            text = ""
 79            last_row = 0
 80            for index in index_list:
 81                if len(text) > 0:
 82                    if last_row == index.row():
 83                        text += "\t"
 84                    else:
 85                        text += os.linesep
 86                text += strip_linebreaks(str(self.model().column(index)))
 87                last_row = index.row()
 88            clipboard.copy(text)
 89
 90    def delete(self) -> None:
 91        """Delete selected rows"""
 92        sel_model = self.selectionModel()
 93        if sel_model:
 94            sel_rows = sel_model.selectedRows()
 95            if sel_rows:
 96                self.model().remove_rows(sel_rows)
 97
 98    def context_menu(self) -> None:
 99        """Display the custom context menu"""
100        if not self.is_empty() and self._context_menu_enable:
101            ctx_menu = QMenu(self)
102            if self._copyable:
103                ctx_menu.addAction("Copy Cells", self.copy)
104            if self._deletable:
105                ctx_menu.addAction("Delete Row", self.delete)
106            if self._clearable:
107                ctx_menu.addSeparator()
108                ctx_menu.addAction("Clear table", self.clear)
109
110            for action in self._custom_menu_actions:
111                check_not_none(action)
112                check_state(len(action) == 3, f"Invalid custom menu action: {action}")
113                check_argument(callable(action[1]), "The action must be callable")
114                if action[2]:
115                    ctx_menu.addSeparator()
116                ctx_menu.addAction(action[0], action[1])
117
118            ctx_menu.exec_(QCursor.pos())

121        """Add a custom menu action item"""
122        self._custom_menu_actions.append((item_text, action, add_separator))

125        """Whether context menu is enabled or not"""

137        """Whether the widget is clearable or not"""

141        """Whether the table view view has data or not"""
142        return not self.model() or self.model().is_empty()
