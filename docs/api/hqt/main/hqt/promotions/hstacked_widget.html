 80            if idx > (self.count() - 1):

 86    def _slide_to_widget(self, widget: QWidget) -> None:
 87        """TODO"""
 88        if self._active:
 89            return
 90
 91        self._active = True
 92        idx = self.currentIndex()
 93        next_idx = self.indexOf(widget)
 94
 95        if idx == next_idx:
 96            self._active = False
 97            return
 98
 99        offset_x, offset_y = self.frameRect().width(), self.frameRect().height()
100        self.widget(next_idx).setGeometry(self.frameRect())
101
102        if self._slide_direction == Qt.Vertical:
103            if idx < next_idx:
104                offset_x, offset_y = 0, -offset_y
105            else:
106                offset_x = 0
107        else:
108            if idx < next_idx:
109                offset_x, offset_y = -offset_x, 0
110            else:
111                offset_y = 0
112
113        pos_next = self.widget(next_idx).pos()
114        pos_current = self.widget(idx).pos()
115        self._pos_current = pos_current
116        offset = QPoint(offset_x, offset_y)
117        self.widget(next_idx).move(pos_next - offset)
118        self.widget(next_idx).show()
119        self.widget(next_idx).raise_()
120
121        anim_group = QParallelAnimationGroup(self, finished=self.animation_done)
122
123        for index, start, end in zip(
124            (idx, next_idx), (pos_current, pos_next - offset), (pos_current + offset, pos_next)
125        ):
126            animation = QPropertyAnimation(
127                self.widget(index),
128                b"pos",
129                duration=self._slide_speed,
130                easingCurve=self._animation_type,
131                startValue=start,
132                endValue=end,
133            )
134            anim_group.addAnimation(animation)
135
136        self._next_idx = next_idx
137        self._cur_idx = idx
138        self._active = True
139        anim_group.start(QAbstractAnimation.DeleteWhenStopped)
140
141    @pyqtSlot()
142    def animation_done(self) -> None:
143        """TODO"""
144        self.widget(self._cur_idx).hide()
145        self.widget(self._cur_idx).move(self._pos_current)
146        self._active = False
147        self.setCurrentIndex(self._next_idx)
