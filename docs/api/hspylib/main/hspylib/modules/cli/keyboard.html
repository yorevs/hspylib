 93    @staticmethod
 94    def kbhit(wait: bool = False, timeout: float = 0) -> bool:
 95        """Return whether there was an ENTER keyboard press."""
 96        while dr_list := select.select([sys.stdin], [], [], timeout):
 97            if not wait:
 98                return dr_list[0] != []
 99            elif dr_list[0]:
100                return True

102    @classmethod
103    def getch(cls, n: int = 1) -> List["Keyboard"]:
104        """Read and return (a) character(s) from a keyboard press.
105        Method will return after ENTER is pressed.
106        """
107        return list(map(cls.of_value, sys.stdin.read(n)))
108
109    @classmethod
110    def digits(cls) -> List["Keyboard"]:
111        """Return all Keyboard digits."""
112        return list(map(cls.of_value, filter(lambda v: str(v).isdigit(), cls.values())))
113
114    @classmethod
115    def letters(cls) -> List["Keyboard"]:
116        """Return all Keyboard letters."""
117        return list(map(cls.of_value, filter(lambda v: str(v).isalpha(), cls.values())))
118
119    @classmethod
120    def wait_keystroke(cls, blocking: bool = True, ignore_error_keys: bool = True) -> Optional["Keyboard"]:
121        """Wait until a keypress is detected."""
122        try:
123            if keystroke := getkey.getkey(blocking):
124                return cls.of_value(keystroke)
125        except (KeyboardInterrupt, AssertionError) as err:
126            if not ignore_error_keys:
127                raise KeyboardInputError(f"Invalid keystroke => {str(err)}") from err
128        except termios.error as err:
129            raise KeyboardInputError("keyboard:: Requires a terminal (TTY)") from err
130        finally:
131            sys.stdin.flush()
132
133        return None

135    def __str__(self) -> str:
136        return self.name
137
138    def __repr__(self) -> str:
139        return str(self)
140
141    @property
142    def val(self) -> str:
143        return str(self.value)
144
145    def isdigit(self) -> bool:
146        """Return True if the keystroke is a digit string, False otherwise."""
147        return str(self.value).isdigit()
148
149    def isalpha(self) -> bool:
150        """Return True if the keystroke is alphabetic string, False otherwise."""
151        return str(self.value).isalpha()
152
153    def isalnum(self) -> bool:
154        """Return True if the keystroke is alpha-numeric string, False otherwise."""
155        return str(self.value).isalnum()
156
157    def ispunct(self) -> bool:
158        """Return True if the keystroke is a punctuation string, False otherwise."""
159        return all(ch in string.punctuation for ch in str(self.value))
